\begin{lstlisting}
 public int[] TwoSum(int[] nums, int target)
 {
     var map = new Dictionary<int, int>();

     for (int i = 0; i < nums.Length; i++)
     {
         var diff = target - nums[i];
         if (map.ContainsKey(diff)) return [map[diff], i];
         map[nums[i]] = i;
     }
     throw new Exception();
 }
